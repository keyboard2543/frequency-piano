   375	        function getAudioContext() {
   376	            if (!audioCtx) {
   377	                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
   378	            }
   379	            return audioCtx;

   382	        // Play tone function (for piano keys - short duration)
   383	        function playTone(frequency, durationMs = 650, type = 'sine') {
   384	            const ctx = getAudioContext();
   385
   386	            const osc = ctx.createOscillator();
   387	            const gain = ctx.createGain();
   388	            const filter = ctx.createBiquadFilter();
   389
   390	            osc.type = type;
   391	            osc.frequency.value = frequency;
   392
   393	            // Low pass filter for softer sound
   394	            filter.type = 'lowpass';
   395	            filter.frequency.value = 1800;
   396
   397	            // ADSR-like envelope
   398	            const now = ctx.currentTime;
   399	            gain.gain.value = 0.0001;
   400
   401	            // Attack
   402	            gain.gain.linearRampToValueAtTime(0.35, now + 0.025);
   403	            // Decay
   404	            gain.gain.linearRampToValueAtTime(0.22, now + 0.12);
   405	            // Release
   406	            gain.gain.linearRampToValueAtTime(0.0001, now + (durationMs / 1000));
   407
   408	            osc.connect(filter);
   409	            filter.connect(gain);
   410	            gain.connect(ctx.destination);
   411
   412	            osc.start(now);
   413	            osc.stop(now + (durationMs / 1000) + 0.3);
   414
   415	            // Cleanup
   416	            setTimeout(() => {
   417	                try { osc.disconnect(); } catch(e) {}
   418	            }, durationMs + 600);
   419	        }

   421	        // Piano keys data (C4 to C6)
   422	        let pianoKeys = [];
   423
   424	        function generatePianoKeys() {
   425	            pianoKeys = [];
   426	            const noteNames = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'A#', 'B'];
   427
   428	            for (let midi = 36; midi <= 108; midi++) { // C2 (36) to C8 (108)
   429	                const freq = 440 * Math.pow(2, (midi - 69) / 12);
   430	                const noteIndex = midi % 12;
   431	                const octave = Math.floor(midi / 12) - 1;
   432	                const noteName = noteNames[noteIndex] + octave;
   433	                const isBlack = [1, 3, 6, 8, 10].includes(noteIndex);
   434
   435	                pianoKeys.push({
   436	                    midi: midi,
   437	                    freq: parseFloat(freq.toFixed(2)),
   438	                    noteName: noteName,
   439	                    isBlack: isBlack
   440	                });
   441	            }
   442	        }

   583	                if (keyEl !== activeGlissandoKey) {
   584	                    activeGlissandoKey = keyEl;

   660	        function toggleFrequencyPlay() {
   661	            const btn = document.getElementById('freq-play-btn');
   662	            const icon = document.getElementById('play-icon');
   663	            const text = document.getElementById('play-text');
   664	            const input = document.getElementById('freq-input');
   665	            const status = document.getElementById('freq-status');
   666
   667	            const freq = parseFloat(input.value);
   668	            if (!freq || freq < 20 || freq > 20000) {
   669	                alert('กรุณาป้อนความถี่ระหว่าง 20 - 20000 Hz');
   670	                return;
   671	            }
   672
   673	            const ctx = getAudioContext();
   674
   675	            if (!isFreqPlaying) {
   676	                // START playing
   677	                if (currentOsc) {
   678	                    try { currentOsc.stop(); } catch(e) {}
   679	                }
   680
   681	                currentOsc = ctx.createOscillator();
   682	                currentGain = ctx.createGain();
   683	                const filter = ctx.createBiquadFilter();
   684
   685	                currentOsc.type = 'sine';
   686	                currentOsc.frequency.value = freq;
   687
   688	                filter.type = 'lowpass';
   689	                filter.frequency.value = 2200;
   690
   691	                currentGain.gain.value = 0.4;
   692
   693	                currentOsc.connect(filter);
   694	                filter.connect(currentGain);
   695	                currentGain.connect(ctx.destination);
   696
   697	                currentOsc.start();
   698
   699	                isFreqPlaying = true;
   700
   701	                // Update UI
   702	                btn.classList.remove('bg-emerald-600', 'hover:bg-emerald-500');
   703	                btn.classList.add('bg-rose-600', 'hover:bg-rose-500');
   704	                icon.classList.remove('fa-play');
   705	                icon.classList.add('fa-stop');
   706	                const t2 = translations[currentLang];
   707	                text.textContent = t2.stopSound;
   708
   709	                updateFreqStatus(freq);
   710
   711	                // Live update listener
   712	                setupLiveUpdate();
   713	            } else {
   714	                // STOP playing
   715	                stopFrequencyTone();
   716	            }
   717	        }

   747	                newSlider.value = Math.min(Math.max(parseFloat(newInput.value) || 20, 20), 4000);
